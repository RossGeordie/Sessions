let
    Source = Json.Document(Web.Contents(<<API>>)),

    // Convert list of records into a table
    AsTable = Table.FromList(Source, Splitter.SplitByNothing(), null, null, ExtraValues.Error),

    // Each row is a record in Column1
    WithPrice = Table.AddColumn(AsTable, "price", each try [Column1][price] otherwise null),
    WithName  = Table.AddColumn(WithPrice, "name",  each try [Column1][name]  otherwise null),
    WithId    = Table.AddColumn(WithName, "id",    each try [Column1][id]    otherwise null),

    // Expand only guaranteed fields
    Expanded = Table.ExpandRecordColumn(WithId, "Column1", {"created_at","updated_at","state"})
in
    Expanded
