let
    Source = PreviousStep,   // your table with column "Incidents"

    // Convert each nested table to a record (first row if only one record per incident)
    AsRecord = Table.AddColumn(
        Source,
        "IncidentRecord",
        each if Table.IsEmpty([Incidents]) 
             then null 
             else Record.FromTable([Incidents]{0})
    ),

    // Safely extract fields
    WithId    = Table.AddColumn(AsRecord, "id",    each Record.FieldOrDefault([IncidentRecord], "id", null)),
    WithName  = Table.AddColumn(WithId,    "name",  each Record.FieldOrDefault([IncidentRecord], "name", null)),
    WithPrice = Table.AddColumn(WithName,  "price", each Record.FieldOrDefault([IncidentRecord], "price", null)),

    Result = Table.RemoveColumns(WithPrice, {"Incidents","IncidentRecord"})
in
    Result
